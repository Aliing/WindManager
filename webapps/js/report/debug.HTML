Asynchronous ReST/JSON API in generic
  1. Browser example 
    2.     <form name=ExampleForm action=/CONTEXT/SERVICE/OPERATION/REQUESTER/REQUEST?revision=...

    AJAX.post
    ({
      form: document.ExampleForm,
      handleAs: "json",
      load: function (JSON)
      {
        switch( JSON.result )
        {
          case 0:
            warning = JSON.exception;
            break;
          default:
            error = JSON.exception;
        }
        ...
      }
    });
  2. Service & operation, optional revision

    1. Service string identifies a particular service, for instance, "report" for this reporting service.

    2. Operation string identifies a particular operation, for instances, "data" for Report Data retrieval operation.

    3. Revision integer identifies service & operation revision. It's a 16-bit signed integer, optionally prefixing w/ 0x/0X/# for hexa-decimal or 0 for octal after optional sign.

  3. Optional requester & request, to retrieve response asynchronously, or after Time Out or the particular responding (cluster) element crashed/restarted, @ a different trip

    1. Requester string scopes the request identifier.

    2. Request integer identifies the corresponding response to be retrieved asynchronously. It's a 64-bit signed integer, optionally prefixing w/ 0x/0X/# for hexa-decimal or 0 for octal after optional sign. E.g. new Date().getTime()

  4. Response JSON

    1. Result integer

    2. Exception string for support

  5. Localized expression validation (/hm/js/report/data.js) 
    6. AeroHive =
{
  metric: function (display)
  {
    return '[' + display + ']';
  },

  metrics: function (report,metric)
  {
    for( index = 0 ; report.length != index ; ++index )
    {
      var metrics = report[ index ];
      if( REPORT_DATA == report )
        metrics = metrics[0];
      if( AeroHive.axis( metrics) == metric )
        return report[ index ];
    }
  },
  localization: function (axis)
  {
    return AeroHive.metrics( REPORT_LOCALIZATION, axis);
  },

  replace: function (expression,global,replacement)
  {
    for( replaced = expression.replace( global, replacement) ; replaced != expression ;)
      replaced = (expression = replaced).replace( global, replacement);
    return expression;
  },
  validate: function (expression,axis)
  {
    var row = AeroHive.localization( axis);
    for( metric in row )
    {
      var display = AeroHive.metric( row[ metric ].display);
      expression = AeroHive.replace
      (
        AeroHive.replace( expression, "AH(" + display, 'AH(rows,row,"' + metric + '"'),
        display,
        'rows[row]["' + metric + '"]'
      );
    }
    var rows = AeroHive.metrics( REPORT_DATA, axis);
    row = 0;
    return undefined == eval( expression)
         ? undefined
         : expression;
